6) Considere os seguintes nomes de variáveis e diga quais deles são ilegais e porquê.

    a) var NOME;

    b) var $num1;

    c) var typeof;

    A palavra typeof é reservada pelo js

    d) var nome-aluno;

    Não é permitido o uso de - traço ao declarar uma variável

    e) var tentativa_2;

    f) var 2a_tentativa;

    Não é permitido iniciar as variaveis com números

    g) var nome completo;

    Não é permitido declarar variaveis com espaço


7) Considere o bloco de código abaixo e responda os valores que aparecerão no console. 

                    

                    var quantidade = "25";
                    var numero = 6;
                    var pressao;
                    var temperatura = 40;
                    temperatura = null;

                    console.log(quantidade += quantidade);
                    // essa função retornará o número 2525

                    console.log( (7+5) / numero + 2 ); 
                    // essa função retornará o número '4'

                    console.log(pressao);
                    // essa função retornará o número 'undefined'

                    console.log(temperatura);
                    // essa função retornará 'null'

                    console.log(typeof pressao);
                    // essa função retornará 'undefined'

                    console.log(typeof temperatura);
                    // essa função retornará 'object'

8) Indique o que será retornado no console nas comparações abaixo. 

                    

                    var idade = 65;

                    console.log(idade != 65);
                    // essa função retornará 'false'

                    console.log(idade >= 65); 
                    // essa função retornará 'true'
                    
                    console.log("65" == idade);
                    // essa função retornará 'true'
                    
                    console.log(65 !== idade);
                    // essa função retornará 'false'
                    
                    console.log(typeof (idade > 60));

                    console.log(typeof (idade > 70));
